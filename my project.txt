  m.volume=0.3;
  m.play();

function typeText(text){
 let i=0;
 let q=document.getElementById("question");
 q.innerHTML="";
 let interval=setInterval(()=>{
  q.innerHTML+=text.charAt(i);
  i++;
  if(i>=text.length) clearInterval(interval);
 },40);
}

 }
}
